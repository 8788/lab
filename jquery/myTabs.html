==== html结构 ====
<div class="tabs">
	<ul class="tabs-title cls">
		<li class="active"><a href="javascript:;">选项卡1</a></li>
		<li><a href="javascript:;">选项卡2</a></li>
		<li><a href="javascript:;">选项卡3</a></li>
	</ul>

		<div class="option">选项卡1的内容</div>

==== jquery代码 ====
;(function(){
	$.fn.myTabs = function(options){
		var opts = $.extend({}, $.fn.myTabs.defaults, options);
		return this.each(function(){
			var that = $(this);
			that.find('.tabs-title li').bind(opts.eventType, function(){
				that.find('.tabs-title li').removeClass('active');
				$(this).addClass('active');
				that.find('.option').hide().eq($(this).index()).show();
			});
		});
	}
	$.fn.myTabs.defaults = {
		eventType: 'mouseover'
	}
}(jQuery));	
$('.tab1').myTabs();
$('.tab2').myTabs({eventType:'click'}); //此处可以自定义事件类型，默认为mouseover
